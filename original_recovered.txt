// ─── Scoring Engine ──────────────────────────────────────────────────────────────

function computeScores(form: FormState): DiseaseScore[] {
  const s: Record<string, number> = {};
  DISEASES.forEach(d => s[d.id] = 0);

  const age = parseInt(form.ageMonths) || 0;
  const c   = form.complaints || [];
  const hpi = form.hpi || {} as HpiData;
  const pmh = form.pmh || {} as PmhData;
  const fam = form.family || {} as FamilyData;
  const v   = form.vitals || {} as VitalsData;
  const ros = form.ros || {} as RosData;

  if (c.includes('cough'))                { s.pneumonia+=2;s.tuberculosis+=2;s.asthma+=2;s.bronchiolitis+=1;s.urti+=2;s.bronchiectasis+=1;s.foreign_body+=1; }
  if (c.includes('fever'))                { s.pneumonia+=3;s.urti+=2;s.epiglottitis+=2;s.tuberculosis+=1;s.empyema+=2;s.retropharyngeal+=2; }
  if (c.includes('wheeze'))               { s.asthma+=4;s.bronchiolitis+=3;s.foreign_body+=1;s.chf+=1;s.anaphylaxis+=1; }
  if (c.includes('stridor'))              { s.croup+=4;s.epiglottitis+=4;s.foreign_body+=3;s.anaphylaxis+=2;s.retropharyngeal+=2; }
  if (c.includes('difficulty_breathing')) { s.pneumonia+=2;s.asthma+=2;s.bronchiolitis+=2;s.chf+=2;s.pleural_effusion+=1;s.pneumothorax+=2;s.anaphylaxis+=1; }
  if (c.includes('hemoptysis'))           { s.tuberculosis+=4;s.bronchiectasis+=3;s.pleural_effusion+=1; }
  if (c.includes('chest_pain'))           { s.pleural_effusion+=3;s.pneumonia+=1;s.pneumothorax+=3;s.chf+=1; }
  if (c.includes('night_sweats'))         { s.tuberculosis+=5; }
  if (c.includes('weight_loss'))          { s.tuberculosis+=5;s.bronchiectasis+=2; }
  if (c.includes('cyanosis'))             { s.pneumonia+=2;s.asthma+=2;s.chf+=3;s.bronchiolitis+=2;s.pneumothorax+=2; }
  if (c.includes('reduced_feeding') && age < 24) { s.bronchiolitis+=3;s.pneumonia+=2; }
  if (c.includes('lethargy'))             { s.pneumonia+=2;s.tuberculosis+=1; }
  if (c.includes('sore_throat'))          { s.urti+=4;s.epiglottitis+=2;s.retropharyngeal+=2; }
  if (c.includes('nasal_discharge'))      { s.urti+=4;s.bronchiolitis+=1;s.allergic_rhinitis+=3;s.sinusitis+=2; }
  if (c.includes('chest_tightness'))      { s.asthma+=3; }
  if (c.includes('noisy_breathing'))      { s.croup+=2;s.foreign_body+=2;s.epiglottitis+=1; }
  if (c.includes('vomiting_symptom'))     { s.pneumonia+=1;s.urti+=1; }
  if (c.includes('diarrhea_symptom'))     { s.urti+=1; }
  if (c.includes('seizure_symptom'))      { s.pneumonia+=1; }
  if (c.includes('rash_symptom'))         { s.urti+=1;s.anaphylaxis+=2; }
  if (c.includes('ear_pain'))             { s.urti+=2;s.retropharyngeal+=1; }
  if (c.includes('abdominal_pain'))       { s.pneumonia+=1; }

  if (hpi.coughChar === 'barking')         s.croup += 6;
  if (hpi.coughChar === 'productive')     { s.pneumonia += 3; s.bronchiectasis += 3; s.empyema += 2; }
  if (hpi.coughChar === 'whooping')        s.urti += 2;
  if (hpi.coughChar === 'blood_stained')  { s.tuberculosis += 3; s.bronchiectasis += 3; }
  if (hpi.coughChar === 'dry')            { s.asthma += 2; s.allergic_rhinitis += 1; }
  if (hpi.coughDuration === 'chronic')    { s.tuberculosis += 4; s.bronchiectasis += 3; s.asthma += 2; }
  if (hpi.coughDuration === 'acute')      { s.urti += 2; s.pneumonia += 1; }
  if (hpi.coughDuration === 'subacute')   { s.pneumonia += 2; s.tuberculosis += 1; }
  if (hpi.nocturnalCough)                 { s.asthma += 3; }
  if (hpi.exerciseTriggered)              { s.asthma += 4; }
  if (hpi.allergenTrigger)                { s.asthma += 3; s.allergic_rhinitis += 2; }
  if (hpi.nightSweats)                    { s.tuberculosis += 5; }
  if (hpi.weightLoss)                     { s.tuberculosis += 5; s.bronchiectasis += 2; }
  if (hpi.tbContact)                      { s.tuberculosis += 6; }
  if (hpi.sickContact)                    { s.pneumonia += 1; s.urti += 1; }
  if (hpi.suddenOnset)                    { s.foreign_body += 5; s.anaphylaxis += 2; }
  if (hpi.unilateralWheeze)               { s.foreign_body += 5; }
  if (hpi.wheezePattern === 'recurrent')  { s.asthma += 5; }
  if (hpi.wheezePattern === 'first' && age <= 24) { s.bronchiolitis += 4; }
  if (hpi.drooling)                       { s.epiglottitis += 5; s.retropharyngeal += 3; }
  if (hpi.tripodPosition)                 { s.epiglottitis += 4; }
  if (hpi.highFever)                      { s.epiglottitis += 3; s.pneumonia += 2; s.empyema += 2; }
  if (hpi.feedingDiff && age <= 12)       { s.bronchiolitis += 3; }
  if (hpi.hepatomegalyReported)           { s.chf += 5; }
  if (hpi.chestIndrawing)                 { s.pneumonia += 3; s.asthma += 2; }
  if (hpi.grunting)                       { s.pneumonia += 4; }
  if (hpi.nasalFlaring)                   { s.pneumonia += 2; s.bronchiolitis += 2; }
  if (hpi.seizureHPI)                     { s.pneumonia += 1; }

  if (pmh.asthmaDx)                       { s.asthma += 6; }
  if (pmh.prevWheeze)                     { s.asthma += 3; }
  if (pmh.recurrentChest)                 { s.bronchiectasis += 4; s.tuberculosis += 2; }
  if (pmh.cardiacDisease)                 { s.chf += 5; }
  if (pmh.hiv)                            { s.pneumonia += 2; s.tuberculosis += 3; }
  if (pmh.sickleCellDisease)              { s.pneumonia += 2; }
  if (pmh.immunodeficiencyDx)             { s.pneumonia += 2; s.tuberculosis += 2; }
  if (pmh.cysticFibrosisDx)               { s.bronchiectasis += 5; s.pneumonia += 2; }

  if (fam.tbHousehold)                    { s.tuberculosis += 5; }
  if (fam.asthmaFamily)                   { s.asthma += 3; }
  if (fam.atopyFamily)                    { s.asthma += 2; s.allergic_rhinitis += 2; }
  if (fam.housingConditions === 'crowded') { s.tuberculosis += 2; s.urti += 1; }
  if (fam.smokingExposure)                { s.asthma += 2; s.pneumonia += 1; }

  if (age <= 24) s.bronchiolitis += 1;
  if (age > 24)  s.bronchiolitis = Math.max(0, s.bronchiolitis - 2);
  if (age < 60)  s.foreign_body += 1;
  if (age >= 60) s.asthma += 1;

  const spo2 = parseFloat(v.spo2), temp = parseFloat(v.temp), rr = parseFloat(v.rr);
  if (!isNaN(spo2) && spo2 < 92) { s.pneumonia+=3;s.asthma+=2;s.bronchiolitis+=2;s.chf+=1;s.pneumothorax+=1; }
  if (!isNaN(temp) && temp >= 38.5) { s.pneumonia+=2;s.urti+=1;s.empyema+=2; }
  const highRR = !isNaN(rr) && ((age<12&&rr>=50)||(age<60&&rr>=40)||(age>=60&&rr>=30));
  if (highRR) { s.pneumonia+=2;s.bronchiolitis+=1; }

  if (v.examTrachealDeviation) { s.pneumothorax += 5; s.pleural_effusion += 2; }
  if (v.examDullness) { s.pleural_effusion += 4; s.empyema += 3; }
  if (v.examReducedBS) { s.pleural_effusion += 3; s.pneumothorax += 3; }

  if (ros.peripheralEdema)               { s.chf += 3; }
  if (ros.pallor)                        { s.tuberculosis += 1; }
  if (ros.clubbing)                      { s.bronchiectasis += 3; s.tuberculosis += 1; s.chf += 1; }
  if (ros.petechiae)                     { s.pneumonia += 1; }

  const total = Math.max(Object.values(s).reduce((a,b)=>a+Math.max(b,0),0), 0.01);
  return DISEASES.map((d,i)=>({
    id:d.id, name:d.name, abbr:d.abbr,
    score: Math.max(s[d.id],0),
    pct: Math.round((Math.max(s[d.id],0)/total)*100),
    color: DC[i],
  })).sort((a,b)=>b.score-a.score).filter(d=>d.score>0).slice(0,7);
}

// ─── Severity Engine ─────────────────────────────────────────────────────────────

function getSeverity(form: FormState): SeverityResult | null {
  const c = form.complaints || [], hpi = form.hpi || {} as HpiData, v = form.vitals || {} as VitalsData;
  const spo2 = parseFloat(v.spo2), temp = parseFloat(v.temp);
  if (c.includes('cyanosis') || (!isNaN(spo2)&&spo2<85) || (hpi.drooling&&hpi.tripodPosition) || (hpi.suddenOnset&&c.includes('stridor')))
    return { level:'CRITICAL', color:'#dc2626', bg:'#fef2f2', border:'#fca5a5', msg:'Immediate intervention required — stabilize airway and breathing urgently.' };
  if ((!isNaN(spo2)&&spo2<92) || hpi.grunting || hpi.chestIndrawing || (!isNaN(temp)&&temp>=39.5))
    return { level:'SEVERE', color:'#ea580c', bg:'#fff7ed', border:'#fdba74', msg:'Urgent assessment needed — consider escalation to higher level of care.' };
  if ((!isNaN(spo2)&&spo2<=94) || c.includes('difficulty_breathing') || (!isNaN(temp)&&temp>=38.5))
    return { level:'MODERATE', color:'#d97706', bg:'#fffbeb', border:'#fde68a', msg:'Close monitoring required — initiate targeted treatment.' };
  if (c.length > 0)
    return { level:'MILD', color:'#16a34a', bg:'#f0fdf4', border:'#86efac', msg:'Clinically manageable — may consider outpatient treatment if stable.' };
  return null;
}

// ─── Helper: format age ─────────────────────────────────────────────────────────

function formatAge(ageMonths: string): string {
  const age = parseInt(ageMonths) || 0;
  if (age < 12) { return age + '-month-old'; }
  const years = Math.floor(age / 12);
  const months = age % 12;
  if (months > 0) { return years + '-year ' + months + '-month-old'; }
  return years + '-year-old';
}

// ─── Clinical Note Generator ─────────────────────────────────────────────────────

  }
  if (hpi.site) { hpiNarr += 'Site of symptoms: ' + hpi.site + '. '; }
  if (hpi.character) { hpiNarr += 'Character: ' + hpi.character + '. '; }

  if (hpi.associated) { hpiNarr += 'Associated symptoms: ' + hpi.associated + '. '; }
  if (hpi.timeCourse) { hpiNarr += 'Time course: ' + hpi.timeCourse + '. '; }
  if (hpi.exacerbating) { hpiNarr += 'Exacerbating factors: ' + hpi.exacerbating + '. '; }
  if (hpi.relieving) { hpiNarr += 'Relieving factors: ' + hpi.relieving + '. '; }
  if (hpi.nightSweats) hpiNarr += 'Night sweats are present. ';
  if (hpi.weightLoss)  hpiNarr += 'Weight loss has been noted. ';
  if (hpi.tbContact)   hpiNarr += 'There is a significant history of TB contact. ';
  if (hpi.sickContact) hpiNarr += 'There is known sick contact exposure' + (hpi.sickContactDetail ? ' (' + hpi.sickContactDetail + ')' : '') + '. ';
  if (hpi.suddenOnset) hpiNarr += 'Sudden onset raises concern for foreign body aspiration. ';
  if (hpi.drooling)    hpiNarr += 'Drooling is present — epiglottitis must be urgently excluded. ';

  if (hpi.vomitingHPI) hpiNarr += 'Vomiting has occurred. ';
  if (hpi.diarrheaHPI) hpiNarr += 'Diarrhea has been present. ';
  if (hpi.exposureCovid) hpiNarr += 'There is history of COVID/infectious exposure' + (hpi.exposureCovidDetail ? ' (' + hpi.exposureCovidDetail + ')' : '') + '. ';
  if (hpi.prevTx)      hpiNarr += 'Has received ' + hpi.prevTx + ' prior to this presentation with ' + (hpi.txResponse || 'unspecified') + ' response. ';

  const vRows: string[][] = [
    ['Temperature', v.temp ? v.temp + 'C' + (parseFloat(v.temp)>=38.5 ? ' FEVER' : '') : 'Not recorded'],
    ['Heart Rate', v.hr ? v.hr + ' bpm' : 'Not recorded'],
    ['Respiratory Rate', v.rr ? v.rr + ' breaths/min' : 'Not recorded'],
    ['SpO2', v.spo2 ? v.spo2 + '%' + (parseFloat(v.spo2)<92 ? ' HYPOXIA' : '') : 'Not recorded'],
    ['Blood Pressure', v.bp || (v.bpSystolic && v.bpDiastolic ? v.bpSystolic + '/' + v.bpDiastolic + ' mmHg' : 'Not recorded')],
    ['Weight', v.weight ? v.weight + ' kg' : 'Not recorded'],
    ['Height/Length', v.height ? v.height + ' cm' : 'Not recorded'],
    ['MUAC', v.muac ? v.muac + ' cm' : 'Not recorded'],
    ['Head Circumference', v.hc ? v.hc + ' cm' : 'Not recorded'],
    ['Hydration Status', v.hydration || 'Not recorded'],
    ['General Condition', v.generalCondition ? v.generalCondition.replace(/_/g,' ') : 'Not recorded'],
    ['AVPU', v.avpu || 'Not recorded'],
  ].filter(function(r) { return r[1] !== 'Not recorded'; });

  // Examination text
  var examText = '';
  var respFindings = [
    v.examIndrawing && 'chest indrawing',
    v.examNasalFlaring && 'nasal flaring',
    v.examGrunting && 'expiratory grunting',
    v.examStridor && 'inspiratory stridor',
    v.examWheeze && 'wheeze on auscultation',
    v.examCrackles && 'crackles on auscultation',
    v.examReducedBS && 'reduced breath sounds',
    v.examBronchial && 'bronchial breath sounds',
    v.examDullness && 'dullness on percussion',
    v.examHyperResonance && 'hyper-resonance on percussion',
    v.examTrachealDeviation && 'tracheal deviation',
  ].filter(Boolean);
  if (respFindings.length) { examText += 'Respiratory examination reveals: ' + respFindings.join(', ') + '. '; }
  if (v.examCyanosis) { examText += 'Central cyanosis is present. '; }

  var cvsExamText = '';
  if (v.examHeartSounds) { cvsExamText += 'Heart sounds: ' + v.examHeartSounds + '. '; }
  if (v.examMurmur) { cvsExamText += 'Cardiac murmur detected: ' + v.examMurmur + (v.examMurmurGrade ? ' (Grade ' + v.examMurmurGrade + '/6)' : '') + '. '; }
  if (v.peripheralEdema) { cvsExamText += 'Peripheral edema is present. '; }
  if (!cvsExamText) { cvsExamText = 'Cardiovascular examination within normal limits.'; }

  if (v.examAbdDistension) { abdExamText += 'Abdomen is distended. '; }
  if (v.examHepatomegaly) { abdExamText += 'Hepatomegaly is present. '; }
  if (v.examSplenomegaly) { abdExamText += 'Splenomegaly is present. '; }
  if (v.examTenderness) { abdExamText += 'Abdominal tenderness: ' + v.examTenderness + '. '; }

    mgmt.push({ heading:'Acute Asthma', text:'Salbutamol 2.5-5 mg nebulized q20 min x 3 doses (or MDI 4-8 puffs with spacer).\nIpratropium 250 mcg neb q20 min x 3 (add for moderate-severe).\nPrednisolone PO 1-2 mg/kg/day x 3-5 days (max 40 mg).\nIV Magnesium sulphate 25-50 mg/kg over 20 min for life-threatening.\nConsider IV Aminophylline if poor response to above.' });
  if (scores.some(function(s) { return s.id==='croup'&&s.pct>15; }))
    mgmt.push({ heading:'Croup Management', text:'Mild-Moderate: Dexamethasone 0.15-0.6 mg/kg PO/IM single dose.\nSevere (stridor at rest): Nebulized Adrenaline 1:1000 0.5 mL/kg (max 5 mL) + Dexamethasone IV.\nObserve 2-4 hours post-nebulization. Admit if persistent stridor at rest.' });
  if (scores.some(function(s) { return s.id==='tuberculosis'&&s.pct>15; }))
    mgmt.push({ heading:'TB Workup & Management', text:'Full TB workup before initiating treatment. Refer to TB clinic.\nIf confirmed: RHZE x 2 months then RH x 4 months (weight-based dosing).\nContact tracing of household members essential.\nCheck LFTs before and during treatment.' });
  if (scores.some(function(s) { return s.id==='bronchiolitis'&&s.pct>15; }))
    mgmt.push({ heading:'Bronchiolitis', text:'Supportive care — no proven benefit from bronchodilators, steroids, or antibiotics in typical RSV bronchiolitis.\nOxygen to maintain SpO2 >=92%.\nNasogastric feeds or IV fluids if unable to feed adequately.\nHigh-flow nasal oxygen (HFNO) for moderate-severe cases.\nCPAP if escalating respiratory distress.' });
  if (scores.some(function(s) { return s.id==='epiglottitis'&&s.pct>10; }))
    mgmt.push({ heading:'EPIGLOTTITIS — EMERGENCY', text:'Do NOT examine oropharynx. Do NOT agitate child. Call ENT + Anaesthesiology IMMEDIATELY.\nPrepare for intubation in controlled environment (OR/ICU).\nGive IV Ceftriaxone 100 mg/kg/day after airway secured.\nDexamethasone 0.15 mg/kg IV to reduce airway edema.' });
  if (scores.some(function(s) { return s.id==='pneumothorax'&&s.pct>15; }))
    mgmt.push({ heading:'Pneumothorax', text:'If tension: immediate needle decompression (2nd ICS, midclavicular line).\nChest drain insertion (4th/5th ICS, anterior axillary line).\nConfirm with chest X-ray. Admit for observation.' });
  if (scores.some(function(s) { return s.id==='chf'&&s.pct>15; }))
    mgmt.push({ heading:'Heart Failure Management', text:'Furosemide 1 mg/kg IV/PO.\nOxygen to maintain SpO2 >=94%.\nRestrict fluids if appropriate.\nRefer to pediatric cardiology for echocardiogram.' });

  mgmt.push({ heading:'Supportive Care', text:'Antipyretics: Paracetamol 15 mg/kg q6h PRN (max 60 mg/kg/day).\nAdequate nutrition support — NGT if oral intake severely compromised.\nFluid balance monitoring. Strict intake/output charting.\nMaintain IV fluids if NBM or dehydrated.' });
  mgmt.push({ heading:'Monitoring', text:'Vitals 4-hourly minimum (1-2 hourly if severe). Continuous SpO2 monitoring. Daily weight. Strict fluid balance charting. Assess response to treatment every 24-48 hours.' });
  mgmt.push({ heading:'Counseling & Discharge Planning', text:'Explain diagnosis, treatment plan, and expected course to caregivers in simple terms.\nDanger signs requiring immediate return: increased breathing difficulty, cyanosis, reduced consciousness, poor feeding, seizures.\nFollow-up: 3-5 days (mild-moderate) / 24-48 hours (severe).\nVaccination catch-up if applicable. Stop smoking exposure at home.' });

  const ddxForNote = scores.slice(0,5).map(function(d,i) { return (i+1) + '. ' + d.name + ' ' + d.pct + '% probability'; }).join('\n');

  // Build clinical summary
  var summary = (form.patientName || 'Patient') + ' is a ' + ageStr + ' ' + sex.toLowerCase() + ' ';
  const top = scores[0];
  if (c.length > 0) {
    summary += 'presenting with ' + c.slice(0,3).map(function(x) { return x.replace(/_/g,' '); }).join(', ') + ' ';
  }
  if (v.spo2) { summary += 'with SpO2 of ' + v.spo2 + '%' + (parseFloat(v.spo2)<92 ? ' (significant hypoxia)' : '') + '. '; }
  if (v.rr) { summary += 'Respiratory rate is ' + v.rr + '/min' + '. '; }
  if (v.examIndrawing) { summary += 'Chest indrawing observed. '; }
  if (top) { summary += 'Clinical reasoning favours ' + top.name + ' as the primary diagnosis (' + top.pct + '% probability), '; }
  if (scores[1]) { summary += 'with ' + scores[1].name + ' as the key differential. '; }
  summary += 'Further investigations and clinical correlation are required to confirm the diagnosis and guide management.';

  return {
    ageStr, sex, name, complaintLines, hpiNarr, pmh, birth, dev, imm, nut, fam,
    rosHistory: buildROSHistory(ros),
    vRows, examText, inv: [...inv], mgmt, ddxForNote, scores, clinicalSummary: summary,
    cvsExamText, abdExamText, cnsExamText, mskExamText, skinExamText,
  };
}

function buildROSHistory(ros: RosData): string {
  if (!ros) { return 'No significant findings on review of systems.'; }
  const cns = [ros.seizures&&'seizures',ros.headache&&'headache',ros.lethargyRos&&'lethargy/decreased consciousness',ros.dizziness&&'dizziness',ros.syncope&&'syncope'].filter(Boolean);
  const cvs = [ros.cyanosisRos&&'cyanosis',ros.peripheralEdema&&'peripheral oedema',ros.fatigue&&'easy fatigability',ros.palpitations&&'palpitations'].filter(Boolean);
  const gi  = [ros.vomiting&&'vomiting',ros.diarrhea&&'diarrhoea',ros.abdominalPain&&'abdominal pain',ros.hepatomegaly&&'hepatomegaly',ros.constipation&&'constipation'].filter(Boolean);
  const gu  = [ros.reducedUrine&&'reduced urine output',ros.dysuria&&'dysuria',ros.hematuria&&'hematuria'].filter(Boolean);
  const skin= [ros.rash&&'rash ('+ros.rashType+')',ros.jaundice&&'jaundice',ros.pallor&&'pallor',ros.clubbing&&'digital clubbing',ros.bruising&&'easy bruising',ros.petechiae&&'petechiae'].filter(Boolean);
  const ent = [ros.earPain&&'ear pain',ros.earDischarge&&'ear discharge',ros.soreThroatRos&&'sore throat',ros.nasalDischargeRos&&'nasal discharge',ros.hearingLoss&&'hearing loss'].filter(Boolean);
  const systems: string[] = [];
  if (cns.length)  { systems.push('CNS: ' + cns.join(', ')); }
  if (cvs.length)  { systems.push('CVS: ' + cvs.join(', ')); }
  if (gi.length)   { systems.push('GI: ' + gi.join(', ')); }
  if (gu.length)   { systems.push('GU: ' + gu.join(', ')); }
  if (skin.length) { systems.push('Skin: ' + skin.join(', ')); }
  if (ent.length)  { systems.push('ENT: ' + ent.join(', ')); }
  return systems.length ? systems.join('. ') + '.' : 'No significant findings on review of systems.';
}

// ─── Plain Text Note Builder ─────────────────────────────────────────────────────

function buildNoteText(note: NoteData, form: FormState, severity: SeverityResult | null): string {
  const v = form.vitals || {} as VitalsData;
  const lines: string[] = [];

  lines.push('='.repeat(60));
  lines.push('AMEXAN PEDIATRIC RESPIRATORY CLINICAL NOTE');
  lines.push('='.repeat(60));
  lines.push('Date: ' + new Date().toLocaleDateString('en-GB', { day: '2-digit', month: 'long', year: 'numeric' }));
  lines.push('');

  if (severity) {
    lines.push('SEVERITY: ' + severity.level);
    lines.push(severity.msg);
    lines.push('');
  }

  lines.push('-'.repeat(40));
  lines.push('I. PATIENT BIODATA');
  lines.push('-'.repeat(40));
  lines.push('Name:        ' + note.name);
  lines.push('Age:         ' + note.ageStr);
  lines.push('Sex:         ' + note.sex);
  lines.push('DOB:         ' + (form.dob || 'Not recorded'));
  lines.push('Residence:   ' + (form.residence || 'Not recorded'));
  lines.push('Informant:   ' + (form.informant || 'Not recorded') + ' (' + (form.informantRelation || 'Not recorded') + ')');
  lines.push('Reliability: ' + (form.histReliability || 'Not recorded'));
  lines.push('Admitted:    ' + (form.dateOfAdmission || 'Not recorded'));
  lines.push('Referral:    ' + (form.sourceOfReferral || 'Not recorded'));
  lines.push('');

  lines.push('-'.repeat(40));
  lines.push('II. CHIEF COMPLAINTS');
  lines.push('-'.repeat(40));
  note.complaintLines.forEach(function(l) { lines.push('  * ' + l); });
  if (!note.complaintLines.length) { lines.push('  None recorded'); }
  lines.push('');

  lines.push('-'.repeat(40));
  lines.push('III. HISTORY OF PRESENTING ILLNESS');
  lines.push('-'.repeat(40));
  lines.push(note.hpiNarr);
  lines.push('');

  // Past Medical History
  lines.push('-'.repeat(40));
  lines.push('IV. PAST MEDICAL HISTORY');
  lines.push('-'.repeat(40));
  lines.push(formatPMH(form.pmh));
  lines.push('');

  // Birth History
  lines.push('-'.repeat(40));
  lines.push('V. BIRTH HISTORY');
  lines.push('-'.repeat(40));
  lines.push(formatBirth(form.birth));
  lines.push('');

  // Developmental History
  lines.push('-'.repeat(40));
  lines.push('VI. DEVELOPMENTAL HISTORY');
  lines.push('-'.repeat(40));
  lines.push(formatDev(form.development));
  lines.push('');

  // Immunization History
  lines.push('-'.repeat(40));
  lines.push('VII. IMMUNIZATION HISTORY');
  lines.push('-'.repeat(40));
  lines.push(formatImm(form.immunization));
  lines.push('');

  // Nutritional History
  lines.push('-'.repeat(40));
  lines.push('VIII. NUTRITIONAL HISTORY');
  lines.push('-'.repeat(40));
  lines.push(formatNut(form.nutrition));
  lines.push('');

  // Family & Social History
  lines.push('-'.repeat(40));
  lines.push('IX. FAMILY & SOCIAL HISTORY');
  lines.push('-'.repeat(40));
  lines.push(formatFam(form.family));
  lines.push('');

  // Review of Systems
  lines.push('-'.repeat(40));
  lines.push('X. REVIEW OF SYSTEMS');
  lines.push('-'.repeat(40));
  lines.push(note.rosHistory);
  lines.push('');

  // General Examination
  lines.push('-'.repeat(40));
  lines.push('XI. GENERAL EXAMINATION');
  lines.push('-'.repeat(40));
  note.vRows.forEach(function(r) { lines.push('  ' + r[0] + ': ' + r[1]); });
  if (formatGeneralExam(v)) {
    lines.push('');
    lines.push('  ' + formatGeneralExam(v));
  }
  lines.push('');

  // Systemic Examination
  lines.push('-'.repeat(40));
  lines.push('XII. SYSTEMIC EXAMINATION');
  lines.push('-'.repeat(40));
  lines.push('  Respiratory: ' + note.examText);
  lines.push('  Cardiovascular: ' + note.cvsExamText);
  lines.push('  Abdominal: ' + note.abdExamText);
  lines.push('  CNS: ' + note.cnsExamText);
  lines.push('  Musculoskeletal: ' + note.mskExamText);
  lines.push('  Skin: ' + note.skinExamText);
  lines.push('');

  // Clinical Summary
  lines.push('-'.repeat(40));
  lines.push('XIII. CLINICAL SUMMARY');
  lines.push('-'.repeat(40));
  lines.push(note.clinicalSummary);
  lines.push('');

  // Differential Diagnosis
  lines.push('-'.repeat(40));
  lines.push('XIV. DIFFERENTIAL DIAGNOSIS');
  lines.push('-'.repeat(40));
  lines.push(note.ddxForNote);
  lines.push('  Differential framework:');
  var ddxCategories: string[] = [];
  var c = form.complaints || [];
  if (c.includes('wheeze') || c.includes('wheeze')) { ddxCategories.push('Acute wheeze: Asthma, Bronchiolitis, Foreign body aspiration, Anaphylaxis, Pneumonia'); }
  if (c.includes('cough') && !c.includes('fever')) { ddxCategories.push('Chronic cough: TB, Asthma, Bronchiectasis, Postnasal drip, GERD'); }
  if (c.includes('fever') && c.includes('cough')) { ddxCategories.push('Fever + cough: Pneumonia, TB, Viral URTI, Empyema'); }
  if (c.includes('stridor')) { ddxCategories.push('Stridor: Croup, Epiglottitis, Foreign body, Anaphylaxis, Retropharyngeal abscess'); }
  if (c.includes('difficulty_breathing') || c.includes('cyanosis')) { ddxCategories.push('Respiratory distress: Pneumonia, Asthma, CHF, Bronchiolitis, Pneumothorax'); }
  if (c.some(function(x) { return x==='chest_pain'||x==='hemoptysis'||x==='difficulty_breathing'; })) { ddxCategories.push('Unilateral breath sound reduction: Foreign body aspiration, Pneumothorax, Pleural effusion, Lung collapse'); }
  ddxCategories.forEach(function(cat) { lines.push('  * ' + cat); });
  lines.push('  Note: Probabilities based on clinical data entered. Clinical judgment must guide final diagnosis.');
  lines.push('');

  // Investigations
  lines.push('-'.repeat(40));
  lines.push('XV. SUGGESTED INVESTIGATIONS');
  lines.push('-'.repeat(40));
  note.inv.forEach(function(i) { lines.push('  -> ' + i); });
  lines.push('');

  // Management Plan
  lines.push('-'.repeat(40));
  lines.push('XVI. MANAGEMENT PLAN');
  lines.push('-'.repeat(40));
  note.mgmt.forEach(function(m) {
    lines.push('  ' + m.heading);
    m.text.split('\n').forEach(function(t) { lines.push('    ' + t); });
    lines.push('');
  });

  // Follow-up
  lines.push('-'.repeat(40));
  lines.push('XVII. FOLLOW-UP PLAN');
  lines.push('-'.repeat(40));
  lines.push('  Review in 2-7 days depending on illness severity.');
  lines.push('  Growth monitoring in chronic cases.');
  lines.push('  Return immediately if: increased breathing difficulty, cyanosis, reduced consciousness, poor feeding, seizures.');
  lines.push('');

  lines.push('='.repeat(60));
  lines.push('Generated by AMEXAN Pediatric Respiratory System');
  lines.push('Must be reviewed and adapted to local guidelines before clinical use.');
  lines.push('='.repeat(60));

  return lines.join('\n');
}

// ─── Format Functions ────────────────────────────────────────────────────────────

function formatPMH(pmh: PmhData): string {
  if (!pmh) { return 'Not documented.'; }
  const parts: string[] = [];
  if (pmh.prevAdmissions) { parts.push('Previous hospital admissions: ' + (pmh.prevAdmDetail||'yes')); }
  if (pmh.chronicIllnesses?.length) { parts.push('Chronic illnesses: ' + pmh.chronicIllnesses.join(', ')); }
  if (pmh.allergies) { parts.push('Allergies: ' + pmh.allergies); }
  if (pmh.medications) { parts.push('Current medications: ' + pmh.medications); }
  if (pmh.prevSurgeries) { parts.push('Previous surgeries: ' + (pmh.surgeryDetail||'yes')); }
  if (pmh.asthmaDx) { parts.push('Diagnosed asthma'); }
  if (pmh.prevWheeze) { parts.push('Previous wheeze episodes'); }
  if (pmh.recurrentChest) { parts.push('Recurrent chest infections'); }
  if (pmh.cardiacDisease) { parts.push('Known cardiac disease'); }
  if (pmh.hiv) { parts.push('HIV positive'); }
  if (pmh.sickleCellDisease) { parts.push('Sickle cell disease'); }
  if (pmh.epilepsyDx) { parts.push('Epilepsy'); }
  if (pmh.immunodeficiencyDx) { parts.push('Immunodeficiency'); }
  if (pmh.cysticFibrosisDx) { parts.push('Cystic fibrosis'); }
  if (pmh.diabetesDx) { parts.push('Diabetes mellitus'); }
  if (pmh.downSyndromeDx) { parts.push('Down syndrome'); }
  if (parts.length) { return parts.join('. ') + '.'; }
  return 'No significant past medical history documented.';
}

function formatBirth(birth: BirthData): string {
  if (!birth || (!birth.birthPlace && !birth.gestAge && !birth.deliveryMode && !birth.birthWeight && !birth.neonatalDetail)) {
    return 'Birth history not documented.';
  }
  const parts: string[] = [];
  if (birth.birthPlace) { parts.push('Born at ' + birth.birthPlace); }
  if (birth.gestAge) { parts.push(birth.gestAge + (birth.gestAgeWeeks ? ' at ' + birth.gestAgeWeeks + ' weeks' : '')); }
  if (birth.deliveryMode) { parts.push('delivered by ' + birth.deliveryMode); }
  if (birth.birthWeight) { parts.push('birth weight ' + birth.birthWeight + ' kg'); }
  if (birth.apgar) { parts.push('APGAR: ' + birth.apgar); }
  if (birth.neonatalComplications?.length && !birth.neonatalComplications.includes('None')) {
    parts.push('Neonatal complications: ' + birth.neonatalComplications.join(', '));
  }
  if (birth.nicuAdmission) { parts.push('NICU admission' + (birth.nicuDuration ? ' for ' + birth.nicuDuration : '')); }
  if (birth.neonatalDetail) { parts.push(birth.neonatalDetail); }
  return parts.join(', ') + '.';
}

function formatDev(dev: DevelopmentData): string {
  if (!dev || !dev.grossMotor) { return 'Developmental history not assessed.'; }
  var result = 'Gross motor: ' + dev.grossMotor + (dev.grossMotorDetail ? ' (' + dev.grossMotorDetail + ')' : '');
  result += '. Fine motor: ' + (dev.fineMotor || 'not assessed') + (dev.fineMotorDetail ? ' (' + dev.fineMotorDetail + ')' : '');
  result += '. Speech/language: ' + (dev.speech || 'not assessed') + (dev.speechDetail ? ' (' + dev.speechDetail + ')' : '');
  result += '. Social/emotional: ' + (dev.social || 'not assessed') + (dev.socialDetail ? ' (' + dev.socialDetail + ')' : '');
  result += '.';
  if (dev.concerns) { result += ' Concerns noted: ' + dev.concerns + '.'; }
  return result;
}

function formatImm(imm: ImmunizationData): string {
  if (!imm||!imm.status) { return 'Immunization history not documented.'; }
  var t = 'Immunization status: ' + imm.status.replace(/_/g,' ') + '.';
  if (imm.missedVaccines?.length) { t += ' Missed vaccines: ' + imm.missedVaccines.join(', ') + '.'; }
  if (imm.adverseReactions) { t += ' History of adverse reaction: ' + (imm.adverseDetail||'yes') + '.'; }
  return t;
}

function formatNut(nut: NutritionData): string {
  if (!nut || (!nut.breastfed && !nut.dietaryDiversity && !nut.muac)) { return 'Nutritional history not documented.'; }
  const parts: string[] = [];
  if (nut.breastfed) { parts.push('Breastfeeding: ' + nut.breastfed + (nut.exclusiveDuration ? ' (exclusive for ' + nut.exclusiveDuration + ')' : '') + (nut.bfDuration ? ', total duration ' + nut.bfDuration : '')); }
  if (nut.complementaryAge) { parts.push('Complementary foods introduced at ' + nut.complementaryAge); }
  if (nut.dietaryDiversity) { parts.push('Dietary diversity: ' + nut.dietaryDiversity); }
  if (nut.appetite) { parts.push('Appetite: ' + nut.appetite); }
  if (nut.muac) { parts.push('MUAC: ' + nut.muac + ' cm'); }
  if (nut.weightCentile) { parts.push('Weight centile: ' + nut.weightCentile); }
  if (nut.heightCentile) { parts.push('Height centile: ' + nut.heightCentile); }
  if (nut.malnutritionSigns?.length) { parts.push('Malnutrition signs: ' + nut.malnutritionSigns.join(', ')); }
  return parts.join('. ') + '.';
}

function formatFam(fam: FamilyData): string {
  if (!fam) { return 'Not documented.'; }
  const parts: string[] = [];
  if (fam.tbHousehold) { parts.push('TB contact in household'); }
  if (fam.asthmaFamily) { parts.push('Family history of asthma'); }
  if (fam.atopyFamily) { parts.push('Family history of atopy/eczema/allergic rhinitis'); }
  if (fam.sickleCellFamily) { parts.push('Sickle cell disease in family'); }
  if (fam.epilepsyFamily) { parts.push('Epilepsy in family'); }
  if (fam.similarIllnessSiblings) { parts.push('Similar illness in siblings'); }
  if (fam.geneticDiseases) { parts.push('Genetic conditions: ' + fam.geneticDiseases); }
  if (fam.smokingExposure) { parts.push('Passive smoking exposure' + (fam.smokeDetail ? ' (' + fam.smokeDetail + ')' : '')); }
  if (fam.housingConditions) { parts.push('Housing: ' + fam.housingConditions + (fam.housingDetail ? ' (' + fam.housingDetail + ')' : '')); }
  if (fam.waterSource) { parts.push('Water source: ' + fam.waterSource); }
  if (fam.sanitation) { parts.push('Sanitation: ' + fam.sanitation); }
  if (fam.parentOccupation) { parts.push('Parental occupation: ' + fam.parentOccupation); }
  if (fam.schoolAttendance) { parts.push('School attendance: ' + fam.schoolAttendance + (fam.schoolLevel ? ' (' + fam.schoolLevel + ')' : '')); }
  return parts.length ? parts.join('. ') + '.' : 'No significant family or social history documented.';
}

function formatGeneralExam(v: VitalsData): string | null {
  const parts: string[] = [];
  if (v.generalCondition) { parts.push('General appearance: ' + v.generalCondition.replace(/_/g,' ')); }
  if (v.avpu) { parts.push('AVPU: ' + v.avpu); }
  if (v.hydration) { parts.push('Hydration: ' + v.hydration.replace(/_/g,' ')); }
  if (v.lymphNodes && v.lymphNodes !== 'none') { parts.push('Lymphadenopathy: ' + v.lymphNodes + (v.lymphNodeSite ? ' at ' + v.lymphNodeSite : '')); }
  if (v.pallorExam) { parts.push('Pallor present'); }
  if (v.jaundiceExam) { parts.push('Jaundice present'); }
  if (v.cyanosisExam) { parts.push('Cyanosis present'); }
  if (v.clubbingExam) { parts.push('Digital clubbing present'); }
  if (v.edemaExam) { parts.push('Peripheral edema present'); }
  return parts.length ? parts.join('. ') + '.' : null;
}
